{ ES6 } - Classes

  • ES6 introduced classes.
  • A class is a type of function, but instead of using the keyword function to initiate it, we use the keyword class
  • Use the keyword class to create a class
  • The constructor method is called each time the class object is initialized

{ ES6 } - Classes

            
              class Car {
                constructor(name) {
                  this.brand = name;
                }
              
                present() {
                  return 'I have a ' + this.brand;
                }
              }
              
              class Model extends Car {
                constructor(name, mod) {
                  super(name);
                  this.model = mod;
                }  
                show() {
                    return this.present() + ', it is a ' + this.model
                }
              }
              mycar = new Model("Ford", "Mustang");
              mycar.show();

{ ES6 } Arrow Functions

            
              hello = () => {
                return "Hello World!";
              }

              hello = (val) => "Hello " + val;
              hello = val => "Hello " + val;

              # There is no `this` binding in arrow functions.
              # in Arrow fn `this` represents the object that defined the arrow function

              class Car {
                constructor(name) {
                  this.brand = name;
                }
              
                present = () => {
                  return 'I have a ' + this.brand;
                }
              }

              # you can try this out as a normal function and see if that returns the brand name.

{ ES6 } Variables

            
              
              hello = () => {
                if(1===1) {
                  var a = 10;
                }
                console.log(a); // Would print the value 10
              }

              
              hello = () => {
                if(1===1) {
                  let a = 10;
                }
                console.log(a);  // ReferenceError - Can't find variable 'a'
              }

              
              hello = () => {
                const a = 10;
                a = 20; // TypeError: Attempted to assign to readonly property.
              }
